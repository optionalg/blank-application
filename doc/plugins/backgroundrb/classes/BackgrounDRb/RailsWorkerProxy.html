41:     def run_method host_info,method_name,worker_options = {}
42:       result = []
43:       connection = choose_connection(host_info)
44:       raise NoServerAvailable.new("No BackgrounDRb server is found running") if connection.blank?
45:       if host_info == :local or host_info.is_a?(String)
46:         result << invoke_on_connection(connection,method_name,worker_options)
47:       elsif host_info == :all
48:         succeeded = false
49:         begin
50:           connection.each { |conn| result << invoke_on_connection(connection,method_name,worker_options) }
51:           succeeded = true
52:         rescue BdrbConnError; end
53:         raise NoServerAvailable.new("No BackgrounDRb server is found running") unless succeeded
54:       else
55:         @tried_connections = [connection.server_info]
56:         begin
57:           result << invoke_on_connection(connection,method_name,worker_options)
58:         rescue BdrbConnError => e
59:           connection = middle_man.find_next_except_these(@tried_connections)
60:           @tried_connections << connection.server_info
61:           retry
62:         end
63:       end
64:       return nil if method_name == :ask_work
65:       return_result(result)
66:     end
