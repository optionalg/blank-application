293:             if not @cached_results.nil?
294:                 return @cached_results
295:             end

297:             # Pull out all the results
298:             docs = []
299:             iter = self.matches._begin
300:             while not iter.equals(self.matches._end)
301:                 docs.push({:data => iter.document.data, 
302:                         :percent => iter.percent, 
303:                         :weight => iter.weight,
304:                         :collapse_count => iter.collapse_count})
305:                 iter.next
306:             end
307: 
308:             # Look up without too many SQL queries
309:             lhash = {}
310:             lhash.default = []
311:             for doc in docs
312:                 k = doc[:data].split('-')
313:                 lhash[k[0]] = lhash[k[0]] + [k[1]]
314:             end
315:             # for each class, look up all ids
316:             chash = {}
317:             for cls, ids in lhash
318:                 conditions = [ "#{cls.constantize.table_name}.#{cls.constantize.primary_key} in (?)", ids ]
319:                 found = cls.constantize.find(:all, :conditions => conditions, :include => cls.constantize.xapian_options[:eager_load])
320:                 for f in found
321:                     chash[[cls, f.id]] = f
322:                 end
323:             end
324:             # now get them in right order again
325:             results = []
326:             docs.each{|doc| k = doc[:data].split('-'); results << { :model => chash[[k[0], k[1].to_i]],
327:                     :percent => doc[:percent], :weight => doc[:weight], :collapse_count => doc[:collapse_count] } }
328:             @cached_results = results
329:             return results
